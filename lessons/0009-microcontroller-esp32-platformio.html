บทเรียนที่ 9: การพัฒนาเฟิร์มแวร์ ESP32/ESP8266 แบบไร้รอยต่อ

ในบทเรียนสุดท้ายนี้ เราจะเปลี่ยนการจัดระเบียบโค้ดฮาร์ดแวร์มาสู่ระดับมือโปร นำความรู้เรื่องคลาสและสืบทอด (OOP) มาสลับควบคุมดึงค่าเซ็นเซอร์จริง เรียนรู้ระบบการซ่อนรหัสผ่านให้ปลอดภัย และการสั่งทดสอบไฟกระพริบแบบไม่หน่วงบอร์ดเชื่อมโยงกับ Dashboard

Part 1: secrets.h & .gitignore (การเก็บข้อมูลความลับอย่างปลอดภัย)

ปัญหาที่นักพัฒนาเกือบทุกคนเคยเจอคือการเผลออัปโหลดไฟล์ที่มี SSID, รหัสผ่าน Wi-Fi หรือรหัสผ่าน MQTT ขึ้น GitHub สาธารณะ เราแก้ไขปัญหานี้ได้โดยการแยกไฟล์เก็บค่าความลับ:

1. สร้างไฟล์ชื่อ include/secrets.h ภายใต้โปรเจกต์ PlatformIO:

#ifndef SECRETS_H
#define SECRETS_H

#define WIFI_SSID "ชื่อไวไฟของคุณ"
#define WIFI_PASS "รหัสผ่านไวไฟของคุณ"
#define MQTT_SERVER "192.168.1.100"  // IP ของ Raspberry Pi หรือเซิร์ฟเวอร์
#define MQTT_PORT 1883
#define MQTT_USER "esp_node"
#define MQTT_PASS "รหัสผ่านที่ตั้งไว้ในโบรกเกอร์"

#endif

2. เพิ่มบรรทัดต่อไปนี้ลงในไฟล์ .gitignore ของระบบควบคุมเวอร์ชัน:

# ซ่อนรหัสผ่านและค่าความลับทั้งหมดของบอร์ด
include/secrets.h

เมื่อเราสั่ง Push โค้ดขึ้น Git ตัวสแกนจะเพิกเฉยต่อไฟล์ความลับนี้ ทำให้โปรเจกต์ของคุณปลอดภัย 100%

Part 2: PlatformIO Multi-Board Configuration

ไฟล์ platformio.ini ของโปรเจกต์สามารถประกาศสภาพแวดล้อมรันเพื่อสลับอัปโหลดทั้งบอร์ด **ESP32** และ **ESP8266** ได้พร้อมกัน โดยการกำหนดค่าห้องสมุดไลบรารีที่จำเป็นแยกจากกัน:

; [env:esp32dev] - สำหรับบอร์ด ESP32
[env:esp32dev]
platform = espressif32
board = esp32dev
framework = arduino
monitor_speed = 115200
lib_deps =
    knolleary/PubSubClient @ ^2.8
    bblanchon/ArduinoJson @ ^6.21.3
    adafruit/DHT sensor library @ ^1.4.4
    adafruit/Adafruit Unified Sensor @ ^1.1.9

; [env:nodemcuv2] - สำหรับบอร์ด ESP8266 (NodeMCU)
[env:nodemcuv2]
platform = espressif8266
board = nodemcuv2
framework = arduino
monitor_speed = 115200
lib_deps =
    knolleary/PubSubClient @ ^2.8
    bblanchon/ArduinoJson @ ^6.21.3

Part 3: C++ OOP Sensor Inheritance (สืบทอดและพหุสัณฐานเซ็นเซอร์)

เพื่อให้เห็นการทำงานจริงของการสืบทอด (Inheritance) และการพหุสัณฐาน (Polymorphism) บนฮาร์ดแวร์ เราสร้างคลาสแม่ควบคุมเซ็นเซอร์ทั่วไป และคลาสลูกสำหรับการอ่านค่าจริง:

#include <Arduino.h>

// 1. คลาสแม่ (Abstract Class)
class SensorReader {
protected:
    String sensorName;
    int pin;
public:
    SensorReader(String name, int p) : sensorName(name), pin(p) {}
    virtual void setup() = 0;        // Pure Virtual Function
    virtual float readValue() = 0;   // Pure Virtual Function
    
    String getName() { return sensorName; }
};

// 2. คลาสลูกสำหรับ ESP32: อ่านค่าจาก DHT11 จริง
#ifdef ESP32
#include <DHT.h>
class DHTSensorReader : public SensorReader {
private:
    DHT dht;
    bool isTemperature;
public:
    DHTSensorReader(String name, int p, int dhtType, bool isTemp) 
        : SensorReader(name, p), dht(p, dhtType), isTemperature(isTemp) {}
        
    void setup() override {
        dht.begin();
    }
    
    float readValue() override {
        if (isTemperature) {
            float t = dht.readTemperature();
            return isnan(t) ? -99.0 : t; // ส่งค่า -99.0 หากอ่านค่าไม่สำเร็จ
        } else {
            float h = dht.readHumidity();
            return isnan(h) ? -99.0 : h;
        }
    }
};
#endif

// 3. คลาสลูกสำหรับ ESP8266: อ่านค่าแอนะล็อก A0 (สุ่มค่าทดสอบเมื่อตรวจจับสัญญาณไม่ได้)
class AnalogSensorReader : public SensorReader {
public:
    AnalogSensorReader(String name, int p) : SensorReader(name, p) {}
    
    void setup() override {
        pinMode(pin, INPUT);
    }
    
    float readValue() override {
        int rawVal = analogRead(pin);
        // หากไม่มีอุปกรณ์ต่อเชื่อมจริงๆ ให้สุ่มค่าจำลองเพื่อนวัตกรรม
        if (rawVal == 0 || rawVal == 1024) {
            return (float)random(300, 800) / 10.0;
        }
        return (float)rawVal;
    }
};

// 4. การประยุกต์ใช้งาน Polymorphism ในฟังก์ชันหลัก (main.cpp)
SensorReader* sensor = nullptr;

void setup() {
    Serial.begin(115200);
    
    #ifdef ESP32
    // สร้างออบเจกต์ DHT11 ของ ESP32 ผ่านตัวชี้ของคลาสแม่ (Polymorphism)
    sensor = new DHTSensorReader("esp32_temp", 4, 11, true); // พิน 4, DHT11, อ่านค่า Temp
    #else
    // สร้างออบเจกต์ Analog บน ESP8266
    sensor = new AnalogSensorReader("esp8266_analog", A0);
    #endif

    sensor->setup(); // เรียกใช้ Virtual Method ที่ถูก Override
}

void loop() {
    float value = sensor->readValue();
    Serial.print("Sensor Name: ");
    Serial.print(sensor->getName());
    Serial.print(" | Value: ");
    Serial.println(value);
    
    delay(5000); // ดีเลย์หลัก 5 วินาที
}

Part 4: Non-blocking LED Blinking (สั่งไฟกระพริบโดยไม่ขัดจังหวะบอร์ด)

เมื่อบอร์ด ESP32 ได้รับข้อความสั่งการ "BLINK" มาจากช่องข้อความ MQTT บอร์ดควรสั่งกระพริบ LED จำนวน 5 ครั้ง (สว่าง/ดับ สลับกันห่างกันครั้งละ 500ms) โดยใช้ตัวหน่วงเวลาอิงจาก millis() เพื่อให้ระบบหลักยังสามารถทำงานต่อและเชื่อมต่อเครือข่ายได้แบบไม่บล็อกการทำงาน (Non-blocking):

// กำหนดพินและตัวแปรควบคุม
const int ledPin = 2; // พินไฟ LED ออนบอร์ด
bool isBlinking = false;
int stateCounter = 0;
unsigned long previousBlinkTime = 0;
const int blinkInterval = 500; // 500 มิลลิวินาที

// ฟังก์ชันเริ่มจับจังหวะไฟกระพริบ
void triggerBlinking() {
    isBlinking = true;
    stateCounter = 0;
    previousBlinkTime = millis();
}

// ฟังก์ชันที่จะถูกเรียกใน loop() ตลอดเวลา
void handleLedBlinking() {
    if (!isBlinking) return;
    
    unsigned long currentMillis = millis();
    if (currentMillis - previousBlinkTime >= blinkInterval) {
        previousBlinkTime = currentMillis;
        
        // สลับสถานะไฟเปิด-ปิด (Toggle)
        digitalWrite(ledPin, !digitalRead(ledPin));
        
        stateCounter++;
        // กระพริบ 5 ครั้ง หมายถึงการสลับเปิดปิด 10 ครั้ง
        if (stateCounter >= 10) {
            digitalWrite(ledPin, LOW); // ปิดไฟสนิท
            isBlinking = false;        // สิ้นสุดลำดับการกระพริบ
        }
    }
}

// MQTT Callback ฟังก์ชันเมื่อได้รับคำสั่งเข้ามา
void callback(char* topic, byte* payload, unsigned int length) {
    String msg = "";
    for (int i = 0; i < length; i++) {
        msg += (char)payload[i];
    }
    
    if (msg == "BLINK") {
        triggerBlinking(); // เรียกใช้ฟังก์ชันกระพริบไฟแบบไม่หน่วงเวลาหลัก
    }
}

// 💡 อย่าลืมกำหนดโหมดพินของไฟ LED ในฟังก์ชัน setup()
void setup() {
    pinMode(ledPin, OUTPUT);
    digitalWrite(ledPin, LOW); // ปิดไฟเริ่มต้น
    // ... การตั้งค่าอื่นๆ
}

Part 5: การทดสอบระบบเชื่อมโยงและส่งข้อมูล (Integration Testing)

หลังคอมไพล์โค้ดบอร์ด ESP32 (DHT11) และ ESP8266 (Analog) แล้วเสร็จ ให้ทำการทดสอบระบบบูรณาการร่วมกันตามขั้นตอนนี้:

  1. เปิดระบบบริการ (Docker Compose): สั่งรัน Mosquitto Broker และ FastAPI ในเซิร์ฟเวอร์ Pi หรือคอมพิวเตอร์หลักของคุณด้วยคำสั่ง docker compose up -d
  2. เปิดส่อง Logs ตรวจสอบ: สั่งรัน docker compose logs -f backend เพื่อดูว่า FastAPI ต่อเชื่อมกับ SQLite เรียบร้อยและเตรียมฟังสัญญาณ
  3. จ่ายไฟให้บอร์ดไมโคร: เชื่อมสาย USB หรือป้อนไฟฟ้าให้บอร์ด ส่องหน้าจอ Serial Monitor ใน VS Code (ด้วย Speed 115200) เพื่อยืนยันว่าบอร์ดต่อเชื่อม Wi-Fi และแสดงประโยค "Connected to MQTT Broker!"
  4. ตรวจสอบฐานข้อมูล SQLite: เปิดดูหน้าตารางแดชบอร์ดประวัติในเบราว์เซอร์ของคุณ ยืนยันว่ามีแถวบันทึกข้อมูลเวลาส่งจาก esp32_temp, esp32_humidity และ esp8266_analog ไหลเข้ามาเป็นระยะ
  5. ทดสอบสั่งงานทางไกล (Remote Action): กดปุ่ม "Blink LED" บนแดชบอร์ดหน้าเว็บ สังเกตการแสดงผลทางกายภาพที่ตัวบอร์ด ESP32 ไฟ LED ออนบอร์ดจะต้องเปิดและปิดกระพริบสลับกัน 5 ครั้งอย่างชัดเจน โดยที่ไม่มีข้อมูลเซ็นเซอร์อื่นๆ หล่นหายหรือขาดช่วง

🧠 ตรวจสอบความเข้าใจ (Quick Quiz)

ทำไมเราจึงเลือกใช้วิธีจับเวลาด้วย millis() แทนการใช้คำสั่ง delay(500) เมื่อต้องการสั่งให้ไฟ LED กระพริบในขณะที่บอร์ดรันโปรแกรมระบบเชื่อมต่อ MQTT?

A) เพื่อประหยัดการใช้กระแสไฟฟ้าของบอร์ดไมโครคอนโทรลเลอร์
B) เพื่อป้องกันไม่ให้ CPU บล็อกการประมวลผล ซึ่งจะทำให้โปรเซสเชื่อมโยง MQTT (client.loop) หยุดทำงานจนบอร์ดหลุดสายเชื่อมต่อ
